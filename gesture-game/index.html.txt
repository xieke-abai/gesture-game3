        // --- 生成粒子系统 (基于模型顶点，灰烬消散效果) ---
        function generateParticleSystemFromModel() {
            if (!mainModel) return null;
            // 收集所有顶点 (世界坐标)
            const vertices = [];
            const tempMatrix = new THREE.Matrix4();
            mainModel.updateWorldMatrix(true, false);
            
            mainModel.traverse((child) => {
                if (child.isMesh && child.geometry) {
                    const geometry = child.geometry;
                    const positionAttribute = geometry.attributes.position;
                    if (!positionAttribute) return;
                    const matrix = child.matrixWorld;
                    const positions = positionAttribute.array;
                    for (let i = 0; i < positions.length; i += 3) {
                        const localVec = new THREE.Vector3(positions[i], positions[i+1], positions[i+2]);
                        const worldVec = localVec.clone().applyMatrix4(matrix);
                        vertices.push(worldVec);
                    }
                }
            });
            
            if (vertices.length === 0) {
                // 备用粒子：围绕模型随机生成
                for (let i = 0; i < 800; i++) {
                    vertices.push(new THREE.Vector3(
                        (Math.random() - 0.5) * 2.5,
                        (Math.random() - 0.5) * 3 + mainModel.position.y,
                        (Math.random() - 0.5) * 2 + mainModel.position.z
                    ));
                }
            }
            
            const particleCount = Math.min(3000, vertices.length);
            // 随机选取顶点 (保证数量合适)
            const selectedVerts = [];
            for (let i = 0; i < particleCount; i++) {
                const idx = Math.floor(Math.random() * vertices.length);
                selectedVerts.push(vertices[idx].clone());
            }
            
            const geometry = new THREE.BufferGeometry();
            const positionsArray = new Float32Array(selectedVerts.length * 3);
            selectedVerts.forEach((vec, i) => {
                positionsArray[i*3] = vec.x;
                positionsArray[i*3+1] = vec.y;
                positionsArray[i*3+2] = vec.z;
            });
            geometry.setAttribute('position', new THREE.BufferAttribute(positionsArray, 3));
            
            // 为每个粒子存储速度方向和生命值
            const velocities = [];
            const lifetimes = [];
            for (let i = 0; i < selectedVerts.length; i++) {
                // 速度方向: 向外 + 向上 随机
                const dir = new THREE.Vector3(
                    (Math.random() - 0.5) * 1.2,
                    Math.random() * 1.5 + 0.8,
                    (Math.random() - 0.5) * 1.2
                ).normalize();
                const speed = 0.8 + Math.random() * 1.2;
                velocities.push(dir.multiplyScalar(speed));
                lifetimes.push(1.0); // 初始生命1
            }
            
            // 使用 PointsMaterial 并支持透明度渐变 (整体淡出，并每帧更新位置)
            const particleMaterial = new THREE.PointsMaterial({
                color: 0xccaa88,
                size: 0.12,
                transparent: true,
                opacity: 1.0,
                blending: THREE.AdditiveBlending
            });
            const particleSystem = new THREE.Points(geometry, particleMaterial);
            
            // 附加自定义属性用于动画
            particleSystem.userData = {
                velocities: velocities,
                lifetimes: lifetimes,
                life: 1.0,
                startTime: performance.now()
            };
            
            return particleSystem;
        }
        
        // 灰烬消散动画更新
        let dissolveAnimationId = null;
        function startDissolveAnimation(particleSystem) {
            if (dissolveAnimationId) cancelAnimationFrame(dissolveAnimationId);
            const startTime = performance.now();
            const duration = 2800; // 2.8秒消散
            
            function updateDissolve(now) {
                if (!particleSystem || !particleSystem.parent) {
                    dissolveAnimationId = null;
                    return;
                }
                const elapsed = now - startTime;
                let progress = Math.min(1, elapsed / duration);
                // 更新粒子位置和透明度
                const positionsAttr = particleSystem.geometry.attributes.position;
                const positionsArray = positionsAttr.array;
                const velocities = particleSystem.userData.velocities;
                const delta = 0.016; // 假设每帧约16ms，实际基于时间差更精确? 简单用progress增量位移
                // 更平滑位移: 随时间偏移位置 = 速度 * 时间增量累计
                // 为简单流畅，基于progress计算偏移量: 位移 = 速度 * (progress * 3.5)
                for (let i = 0; i < velocities.length; i++) {
                    const v = velocities[i];
                    const offsetX = v.x * progress * 3.2;
                    const offsetY = v.y * progress * 4.0;
                    const offsetZ = v.z * progress * 2.8;
                    const idx = i * 3;
                    // 原始位置存在userData中? 最好保存原始位置，但我们可以从当前偏移, 更简单: 保存起始点
                    if (!particleSystem.userData.originalPositions) {
                        // 初次运行保存原始位置
                        particleSystem.userData.originalPositions = positionsArray.slice();
                    }
                    const origX = particleSystem.userData.originalPositions[idx];
                    const origY = particleSystem.userData.originalPositions[idx+1];
                    const origZ = particleSystem.userData.originalPositions[idx+2];
                    positionsArray[idx] = origX + offsetX;
                    positionsArray[idx+1] = origY + offsetY;
                    positionsArray[idx+2] = origZ + offsetZ;
                }
                positionsAttr.needsUpdate = true;
                // 透明度渐变
                const opacityVal = Math.max(0, 1 - progress * 1.2);
                particleSystem.material.opacity = opacityVal;
                
                if (progress >= 1.0) {
                    // 动画结束，移除粒子系统，重置模型
                    if (particleSystem.parent) scene.remove(particleSystem);
                    if (particleSystem.geometry) particleSystem.geometry.dispose();
                    if (currentParticleSystem === particleSystem) currentParticleSystem = null;
                    cancelAnimationFrame(dissolveAnimationId);
                    dissolveAnimationId = null;
                    // 重置所有状态回 IDLE，模型复原
                    resetGame();
                    // 确保模型可见且位置正确
                    if (mainModel) mainModel.visible = true;
                    currentState = 'IDLE';
                    updateStatusText('IDLE', '重置完成，张开手掌再次开始');
                } else {
                    dissolveAnimationId = requestAnimationFrame(updateDissolve);
                }
            }
            
            // 保存原始位置快照
            const posAttr = particleSystem.geometry.attributes.position.array;
            particleSystem.userData.originalPositions = posAttr.slice();
            dissolveAnimationId = requestAnimationFrame(updateDissolve);
        }
        
        // --- 手势判断 (MediaPipe landmarks) ---
        // 判断张开手掌：所有手指 (指尖与掌指关节距离足够远) 简化: 检查指尖y坐标相对根部，MediaPipe坐标系中y上小下大
        function isHandOpen(landmarks) {
            if (!landmarks || landmarks.length < 21) return false;
            // 手指尖索引: 4(拇指尖),8(食指尖),12(中指尖),16(无名指尖),20(小指尖)
            // 对应指根: 2,5,9,13,17 简化: 比较指尖y与对应指根y (手竖直张开时指尖y更小)
            const tips = [4, 8, 12, 16, 20];
            const bases = [2, 5, 9, 13, 17];
            let openCount = 0;
            for (let i = 0; i < tips.length; i++) {
                const tipY = landmarks[tips[i]].y;
                const baseY = landmarks[bases[i]].y;
                // 对于掌心朝前，指尖Y坐标小于根部为伸直 (张开)
                if (tipY < baseY - 0.03) openCount++;
            }
            // 至少4个手指伸直视为张开
            return openCount >= 4;
        }
        
        function isHandFist(landmarks) {
            if (!landmarks || landmarks.length < 21) return false;
            const tips = [8, 12, 16, 20];
            const bases = [5, 9, 13, 17];
            let closedCount = 0;
            for (let i = 0; i < tips.length; i++) {
                const tipY = landmarks[tips[i]].y;
                const baseY = landmarks[bases[i]].y;
                // 弯曲状态: 指尖Y大于等于根部或者非常接近
                if (tipY > baseY - 0.02) closedCount++;
            }
            // 拇指单独简单判断
            const thumbTip = landmarks[4];
            const thumbIp = landmarks[3];
            const thumbClosed = thumbTip.x < thumbIp.x; // 握拳时拇指内收
            return (closedCount >= 3) || (closedCount >= 2 && thumbClosed);
        }
        
        // 获取手掌中心y坐标(归一化 0~1)，用于控制模型高度
        function getPalmY(landmarks) {
            if (!landmarks) return 0.5;
            // 使用手腕点(0)和掌心近似平均值
            const wrist = landmarks[0];
            const middleMcp = landmarks[9];
            const avgY = (wrist.y + middleMcp.y) / 2;
            return Math.min(0.95, Math.max(0.05, avgY));
        }
        
        // --- MediaPipe Hands 初始化 ---
        const videoElement = document.getElementById('video-preview');
        let hands = null;
        let cameraUtils = null;
        
        function initMediaPipe() {
            hands = new Hands({
                locateFile: (file) => {
                    return `https://cdn.jsdelivr.net/npm/@mediapipe/hands/${file}`;
                }
            });
            hands.setOptions({
                maxNumHands: 1,
                modelComplexity: 1,
                minDetectionConfidence: 0.5,
                minTrackingConfidence: 0.5
            });
            hands.onResults(onHandResults);
            
            // 使用Camera Utils将摄像头接入MediaPipe
            const cameraObj = new Camera(videoElement, {
                onFrame: async () => {
                    if (hands) await hands.send({image: videoElement});
                },
                width: 640,
                height: 480
            });
            cameraObj.start();
        }
        
        // 手势回调
        function onHandResults(results) {
            if (results.multiHandLandmarks && results.multiHandLandmarks.length > 0) {
                handLandmarks = results.multiHandLandmarks[0];
                isHandDetected = true;
                processGesture();
            } else {
                isHandDetected = false;
                handLandmarks = null;
            }
        }
        
        // 核心手势状态机逻辑
        function processGesture() {
            if (!modelLoaded || !mainModel) return;
            if (currentState === 'DISSOLVING') return; // 消散中不响应
            
            const open = isHandOpen(handLandmarks);
            const fist = isHandFist(handLandmarks);
            
            // 状态机
            if (currentState === 'IDLE') {
                if (open) {
                    currentState = 'ASCENDING';
                    updateStatusText('ASCENDING', '手掌Y轴控制高度 ↑');
                }
            } 
            else if (currentState === 'ASCENDING') {
                // 手掌Y控制模型高度
                if (handLandmarks) {
                    let palmYNorm = getPalmY(handLandmarks);
                    // 手越高(y值越小) 模型越高
                    let targetY = MIN_Y + (1 - palmYNorm) * (MAX_Y - MIN_Y);
                    targetY = Math.min(MAX_Y, Math.max(MIN_Y, targetY));
                    if (mainModel) {
                        mainModel.position.y = targetY;
                    }
                    // 检测是否接近最高高度(>14) 并且 握拳 => 触发靠近悬停
                    if (targetY > 14.0 && fist) {
                        // 结束ASCENDING，开始移动到摄像机前方
                        currentState = 'TRANSITION';
                        updateStatusText('TRANSITION', '握拳! 飞向镜头...');
                        if (moveTween) moveTween.kill();
                        // GSAP 移动模型到 (0, 0, -8)
                        moveTween = gsap.to(mainModel.position, {
                            x: 0, y: mainModel.position.y, z: -8,
                            duration: 0.9,
                            ease: "power2.inOut",
                            onComplete: () => {
                                if (currentState === 'TRANSITION') {
                                    currentState = 'HOVERING_NEAR';
                                    updateStatusText('HOVERING_NEAR', '悬浮中，再次张开手掌→灰烬');
                                    // 添加轻微上下浮动动画
                                    if (floatAnimation) floatAnimation.kill();
                                    floatAnimation = gsap.to(mainModel.position, {
                                        y: mainModel.position.y + 0.22,
                                        duration: 1.2,
                                        repeat: -1,
                                        yoyo: true,
                                        ease: "sine.inOut"
                                    });
                                }
                            }
                        });
                    }
                }
            }
            else if (currentState === 'HOVERING_NEAR') {
                // 检测张开手掌 -> 进入化为灰烬
                if (open) {
                    // 开始 Dissolving
                    currentState = 'DISSOLVING';
                    updateStatusText('DISSOLVING', '灰烬消散...');
                    if (floatAnimation) { floatAnimation.kill(); floatAnimation = null; }
                    if (moveTween) moveTween.kill();
                    // 隐藏模型网格
                    if (mainModel) mainModel.visible = false;
                    // 基于当前模型生成粒子系统
                    const particles = generateParticleSystemFromModel();
                    if (particles) {
                        currentParticleSystem = particles;
                        scene.add(particles);
                        startDissolveAnimation(particles);
                    } else {
                        // 若生成失败，直接重置
                        resetGame();
                        currentState = 'IDLE';
                    }
                }
            }
        }
        
        function updateStatusText(state, action) {
            const statusDiv = document.getElementById('status-text');
            let emoji = '';
            if (state === 'IDLE') emoji = '🌀';
            else if (state === 'ASCENDING') emoji = '📈';
            else if (state === 'HOVERING_NEAR') emoji = '✨';
            else if (state === 'DISSOLVING') emoji = '💨';
            else emoji = '⚡';
            statusDiv.innerHTML = `${emoji} 状态: ${state} | ${action}`;
        }
        
        // --- 按钮事件绑定 ---
        document.getElementById('reset-btn').addEventListener('click', () => {
            if (dissolveAnimationId) cancelAnimationFrame(dissolveAnimationId);
            if (floatAnimation) floatAnimation.kill();
            if (moveTween) moveTween.kill();
            if (currentParticleSystem) {
                scene.remove(currentParticleSystem);
                currentParticleSystem = null;
            }
            resetGame();
            currentState = 'IDLE';
            if (mainModel) mainModel.visible = true;
            updateStatusText('IDLE', '已重置，张开手掌开始');
        });
        
        document.getElementById('fullscreen-btn').addEventListener('click', () => {
            if (!document.fullscreenElement) {
                document.documentElement.requestFullscreen();
            } else {
                document.exitFullscreen();
            }
        });
        
        let camVisible = true;
        document.getElementById('toggle-cam-btn').addEventListener('click', (btn) => {
            const video = document.getElementById('video-preview');
            camVisible = !camVisible;
            video.style.opacity = camVisible ? '1' : '0';
            btn.target.textContent = camVisible ? '📷 隐藏预览' : '📷 显示预览';
        });
        
        // --- 窗口自适应 & 背景平面cover更新 ---
        window.addEventListener('resize', onWindowResize, false);
        function onWindowResize() {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
            if (backgroundPlane && bgTexture) {
                const distance = 45;
                const vFov = (camera.fov * Math.PI) / 180;
                const planeHeight = 2 * Math.tan(vFov / 2) * distance;
                const planeWidth = planeHeight * camera.aspect;
                backgroundPlane.scale.set(planeWidth, planeHeight, 1);
            }
        }
        
        // 辅助动态光源跟随相机 (轻微效果)
        function animateLights() {
            if (fillLight) {
                fillLight.position.x = camera.position.x * 0.3;
                fillLight.position.z = camera.position.z * 0.2 + 2;
            }
        }
        
        // --- 动画循环 ---
        function animate() {
            requestAnimationFrame(animate);
            animateLights();
            // 如果ASCENDING状态没有手势检测，保留当前高度无动作，但不再强制移动
            // 渲染
            renderer.render(scene, camera);
        }
        
        // 启动摄像头与手势
        function startApp() {
            initMediaPipe();
            animate();
            // 初始化背景默认
            setTimeout(() => {
                if (bgTexture) updateBackgroundPlane();
            }, 200);
        }
        
        startApp();
        
        // 简单指示模型未加载完成前的提示
        setTimeout(() => {
            if (!modelLoaded) {
                console.log("等待模型加载，请确保 model.glb 存在");
            }
        }, 1000);
